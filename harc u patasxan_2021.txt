interface Payment {
    public function pay($amount);
}

class CardPayment implements Payment {
    public function pay($amount) {
        echo "Paid $amount by card";
    }
}

class CashPayment implements Payment {
    public function pay($amount) {
        echo "Paid $amount by cash";
    }
}

function makePayment(Payment $payment) {
    $payment->pay(100);
}

makePayment(new CardPayment()); // Paid 100 by card
makePayment(new CashPayment()); // Paid 100 by cash

O — Open/Closed Principle (OCP)

Բաց/փակ սկզբունք

Սկզբունք: Կլասերը պետք է լինեն բաց ընդարձակման համար, բայց փակ փոփոխության համար։
Այսինքն՝ նոր ֆունկցիոնալությունը ավելացնելու համար չպետք է փոխենք արդեն գոյություն ունեցող կոդը, այլ ընդլայնենք այն նոր դասերով կամ ինտերֆեյսներով։

Տերմիններ:

Բաց (Open) – կարելի է ընդարձակել նոր հնարավորություններով

Փակ (Closed) – գոյություն ունեցող կոդը չպետք է փոփոխվի

PHP Օրինակ

interface PaymentMethod {
    public function pay($amount);
}

class CreditCardPayment implements PaymentMethod {
    public function pay($amount) {

L — Liskov Substitution Principle (LSP)

Լիսկովի փոխարինելիության սկզբունք

Սկզբունք: Յուրաքանչյուր ենթադաս պետք է կարողանա փոխարինել իր սուպեր կլասը առանց ծրագրի սխալների ստեղծման։
Դա նշանակում է, որ երեխաները (subclasses) չեն խախտում ծնողի պայմանները։

PHP Օրինակ

class Bird {
    public function fly() { echo "Թռչում է"; }
}

class Sparrow extends Bird {}
class Penguin extends Bird {
    public function fly() {
        throw new Exception("Չի կարող թռչել");
    }
}

Խնդիրը: Penguin չի համապատասխանում LSP-ին, որովհետև չի կարող փոխարինել Bird-ը առանց սխալի։

Լավ լուծում

interface Flyable {
    public function fly();
}

class Sparrow implements Flyable {
    public function fly() { echo "Թռչում է"; }
}

class Penguin {
    public function swim() { echo "Սուզվում է"; }
}

Մտքեր: նախատեսված է, որ subclasses միշտ կարողանան փոխարինել parent class-ը առանց ծրագրի խափանման։

I — Interface Segregation Principle (ISP)

Ինտերֆեյսների առանձնացման սկզբունք

Սկզբունք: Լավ է ունենալ մի քանի փոքր ինտերֆեյս, քան մեկ մեծ ինտերֆեյս։
Կլասը չպետք է ստիպված լինի իրականացնել այն մեթոդները, որոնք նրան պետք չեն։

PHP Օրինակ

interface Worker {
    public function work();
    public function eat();
}

class Robot implements Worker {
    public function work() { echo "Աշխատում է"; }
    public function eat() { /* ռոբոտը չի ուտում */ }
}

Խնդիրը: Robot ստիպված է իրականացնել eat(), չնայած դա չի օգտագործում։

Լավ լուծում

interface Workable {
    public function work();
}

interface Eatable {
    public function eat();
}

class Human implements Workable, Eatable {
    public function work() { echo "Աշխատում է"; }
    public function eat() { echo "Ուտում է"; }
}

class Robot implements Workable {
    public function work() { echo "Աշխատում է"; }
}

Մտքեր: դասերը ստանում են միայն այն ինտերֆեյսները, որոնք իրականում օգտագործում են։

D — Dependency Inversion Principle (DIP)

Մեծ կլասերը չպետք է կախված լինեն փոքրերից։

Երկուսն էլ պետք է կախված լինեն ինտերֆեյսներից (abstractions).

Մանր կլասերը չպետք է կախված լինեն կոնկրետ իրացուցումից, այլ ընդհակառակը։

PHP Օրինակ

// Խախտում՝ կոնկրետ կախվածություն
class MySQLDatabase {
    public function save($data) { /* ... */ }
}

class UserRepository {
    private $db;
    public function __construct() {
        $this->db = new MySQLDatabase();
    }
}

Լավ մոտեցում՝ DIP պահպանելով

interface DatabaseInterface {
    public function save($data);
}

class MySQLDatabase implements DatabaseInterface {
    public function save($data) { /* ... */ }
}

class UserRepository {
    private $db;
    public function __construct(DatabaseInterface $db) {
        $this->db = $db;
    }
}

Մտքեր: UserRepository-ը կախված է abstraction-ից, ոչ կոնկրետ MySQL-ից, հեշտ է փոխարինել այլ DB-ով։

ապահով կապ (encryption)
