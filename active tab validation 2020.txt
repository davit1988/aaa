 public function create()
    {
        $active_tab = 0;
        $errors = session('errors');
        if ($errors) {
            $fields_tabs = [
                ['tab-1'], // Tab 1
                ['title', 'desc'], // Tab 2
                ['avatar', 'bio'], // Tab 3
            ];
            foreach ($fields_tabs as $tab => $fields) {
                foreach ($fields as $field) {
                    if ($errors->has($field)) {
                        $active_tab = $tab;
                        break;
                    }
                }
            }
           
        }
         return view('welcome', compact('active_tab'));
    }
