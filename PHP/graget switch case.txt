function get_array_data( string $type = null )
{
	$typeArray = [

      'apple' => 'this is apple',
      'cow' => 'this is cow',
      'milk' => 'this is milk'

	];

	if (! array_key_exists( $type, $typeArray ) )
    {
     	 return "Not valid type";
    }

	return $typeArray[$type];

}

echo get_array_data('cow');
